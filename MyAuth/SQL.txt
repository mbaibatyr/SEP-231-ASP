create table Users
(
	id int primary key identity,
	[login] varchar(100),
	[psw] varchar(100),
	created datetime default(getdate()),
	[role] varchar(100)
)


create proc pUsers
@login varchar(100),
@psw varchar(100)
as
SELECT id,
	[login],
	[psw],
	created,
	[role]
FROM Users
WHERE [login] = @login
	and psw = @psw

create proc [dbo].[pUsers];2 --'user3', '1234', 'admin'
@login varchar(100),
@psw varchar(100),
@role varchar(100)
as
if exists(
			SELECT *
			FROM Users
			WHERE [login] = @login
		)
		SELECT '1'-- уже существует
else
	begin
		insert into Users (login, psw, role)
		values (@login, @psw, @role)
		select '0'-- успешно добавлен
	end
